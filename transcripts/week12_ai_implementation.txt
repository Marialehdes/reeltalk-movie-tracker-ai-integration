  1. Backend — server.js (Node.js)

  Create this file in your project root. It receives the raw review from the
  browser and forwards it to Claude.

  // server.js
  const http = require("http");
  const https = require("https");

  const PORT = 3001;
  const ANTHROPIC_API_KEY = process.env.ANTHROPIC_API_KEY; // set in your 
  terminal

  const server = http.createServer((req, res) => {
    // Allow requests from your HTML file
    res.setHeader("Access-Control-Allow-Origin", "*");
    res.setHeader("Access-Control-Allow-Headers", "Content-Type");

    if (req.method === "OPTIONS") {
      res.writeHead(204);
      res.end();
      return;
    }

    if (req.method === "POST" && req.url === "/enhance-review") {
      let body = "";
      req.on("data", (chunk) => (body += chunk));
      req.on("end", () => {
        const { review } = JSON.parse(body);

        const payload = JSON.stringify({
          model: "claude-opus-4-6",
          max_tokens: 1024,
          messages: [
            {
              role: "user",
              content: `Polish this rough movie review into a well-written, 
  engaging paragraph. Keep the same opinion and tone — just improve the grammar,
   clarity, and flow. Return only the improved review, nothing else.\n\nReview: 
  ${review}`,
            },
          ],
        });

        const options = {
          hostname: "api.anthropic.com",
          path: "/v1/messages",
          method: "POST",
          headers: {
            "Content-Type": "application/json",
            "x-api-key": ANTHROPIC_API_KEY,
            "anthropic-version": "2023-06-01",
          },
        };

        const apiReq = https.request(options, (apiRes) => {
          let data = "";
          apiRes.on("data", (chunk) => (data += chunk));
          apiRes.on("end", () => {
            const result = JSON.parse(data);
            const enhanced = result.content[0].text;
            res.writeHead(200, { "Content-Type": "application/json" });
            res.end(JSON.stringify({ enhanced }));
          });
        });

        apiReq.on("error", (err) => {
          res.writeHead(500);
          res.end(JSON.stringify({ error: err.message }));
        });

        apiReq.write(payload);
        apiReq.end();
      });
    } else {
      res.writeHead(404);
      res.end();
    }
  });

  server.listen(PORT, () =>
    console.log(`Server running at http://localhost:${PORT}`)
  );

  To run it:
  # In your terminal (set your key first)
  ANTHROPIC_API_KEY=your-key-here node server.js

  3. React Component Version (for future reference)

  Since you asked for it — here's the self-contained React equivalent if you
  ever migrate:

  import { useState } from "react";

  export function ReviewEnhancer() {
    const [rawReview, setRawReview] = useState("");
    const [enhanced, setEnhanced] = useState("");
    const [loading, setLoading] = useState(false);

    async function handleEnhance() {
      if (!rawReview.trim()) return;
      setLoading(true);

      try {
        const res = await fetch("http://localhost:3001/enhance-review", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ review: rawReview }),
        });
        const data = await res.json();
        setEnhanced(data.enhanced);
      } catch (err) {
        console.error(err);
      } finally {
        setLoading(false);
      }
    }

    return (
      <div className="review-enhancer">
        <h2>AI Review Enhancer</h2>
        <textarea
          value={rawReview}
          onChange={(e) => setRawReview(e.target.value)}
          placeholder="Write your rough review here..."
          rows={4}
        />
        <button onClick={handleEnhance} disabled={loading}>
          {loading ? "Enhancing..." : "Enhance Review"}
        </button>

        {enhanced && (
          <div className="enhanced-output">
            <h3>Enhanced Review</h3>
            <p>{enhanced}</p>
          </div>
        )}
      </div>
    );
  }

● Here's what changed and why:

  ---
  What changed

  ┌────────────┬─────────────────────────────────┬──────────────────────────┐
  │            │         Before (Claude)         │      After (Ollama)      │
  ├────────────┼─────────────────────────────────┼──────────────────────────┤
  │ Module     │ http + https                    │ http only — Ollama is    │
  │            │                                 │ local, no TLS needed     │
  ├────────────┼─────────────────────────────────┼──────────────────────────┤
  │ API key    │ ANTHROPIC_API_KEY env var       │ Removed entirely — no    │
  │            │                                 │ auth needed              │
  ├────────────┼─────────────────────────────────┼──────────────────────────┤
  │ Host       │ api.anthropic.com (HTTPS)       │ localhost:11434 (HTTP)   │
  ├────────────┼─────────────────────────────────┼──────────────────────────┤
  │ Endpoint   │ /v1/messages                    │ /api/generate            │
  ├────────────┼─────────────────────────────────┼──────────────────────────┤
  │ Payload    │ { model, max_tokens, messages:  │ { model, prompt, stream: │
  │            │ [{role, content}] }             │  false }                 │
  ├────────────┼─────────────────────────────────┼──────────────────────────┤
  │ Response   │ parsed.content[0].text          │ parsed.response          │
  │ field      │                                 │                          │
  └────────────┴─────────────────────────────────┴──────────────────────────┘

  stream: false — why it's there

  Ollama streams tokens by default (one JSON chunk per token). Setting stream: 
  false makes it wait and return a single complete JSON object, which is much
  simpler to handle.

  How to run it

  Make sure Ollama is running and the model is pulled first:
  ollama pull gemma:2b
  ollama serve          # if it isn't already running

  Then start your server — no environment variables needed:
  node server.js
